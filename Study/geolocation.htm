var locator = window.navigator.geolocation;
	var display = document.querySelector("#display");
		
	document.getElementById("locationButton").addEventListener("click", callGeo );
	
	function callGeo(){
		locator.getCurrentPosition(success, failure);
	}
	
	function success (position){
		var newDiv = document.createElement("div");
		newDiv.textContent = "long: " + position.coords.longitude + " lat: " + position.coords.latitude;
		display.appendChild(newDiv);
		console.log("what the heck");
	}
	
	function failure (e){
		console.log(e.message);
